# Outliers Detection and handling

from scipy.stats import zscore
import numpy as np

z_scores = np.abs(zscore(df[['Runs', 'BF']]))
df_outliers = df[(z_scores < 3).all(axis=1)]  # Keep only rows without outliers

print(f'Shape after removing outliers: {df_outliers.shape}')
